#include <stdio.h>

int main() {
    int t1, t2, t3;
    printf("Введіть три значення:\n");
    scanf("%d %d %d", &t1, &t2, &t3);

    // Розраховуємо загальний час, який потрібен гостям
    float total_time = 1.0 / t1 + 1.0 / t2 + 1.0 / t3;

    // Виводимо результат з округленням до 2 знаків після коми
    
        printf("Час, необхідний для з'їдання пирога %.2f годин\n", 1.0 / total_time);
    return 0;
}
